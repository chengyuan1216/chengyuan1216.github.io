class

  1. 用数组表示
    :class="[red]"
    //red是vue对象的属性数据，data:{ red:"classred" }, classred是提前定义好的类

    :class="[red,b,c,d]" //可以添加多个类
  2. 用json表示
    :class="{red:a, blue:false}"
    //red,blue是定义好的css类， 用true、false作为值决定是否使用样式

    :class="json" //需要在data里先定义好json
    data:{ json:{ red:true, blue:false } } 
        <div id="div1" :class="[name]">

        var vm = new Vue({
            el:"#div1",
            data:{
                name:"red"
            }
        });
        vm.name="blue";
    
        <div id="div1" :class="{red:true,blue:false}">

        var vm = new Vue({
            el:"#div1"
        });
    
        <div id="div1" :class="json">

        var vm = new Vue({
            el:"#div1",
            data:{
                json:{red:false,blue:true}
            }
        });

style

  1. 用数组表示
    :style="[c]"
    :style="[c,d]"//c、d均是json
  2. 用json表示
    :style="json"
    复合样式，采用驼峰命名法
    注意:
    在data定义以json的形式定义样式
        <div id="div2" :style="json">
        </div>

        var sty = {
            width:"300px",
            height:"50px",
            background:"red"
        };

        new Vue({
            el:"#div2",
            data:{
                json:sty
            }
        })
